##############################
# A new event every 10 years until 100 yrs, then one for 150 and 200
##############################

country_event = {
    id = aiboost.100
    hide_window = yes
    is_triggered_only = yes
    immediate = {
        every_country = {
            limit = {
                is_ai = yes
                is_country_type = default
            }
            country_event = { id = aiboost.200 }
        }
        if = {
            limit = {
                years_passed > 10
            }
            country_event = { id = aiboost.101 days = 365 }
            else = {
                country_event = { id = aiboost.100 days = 365 }
            }
        }
    }
}
