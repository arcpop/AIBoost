namespace = aiboost


country_event = {
    id = aiboost.1
    hide_window = yes
    trigger = {
        is_ai = no
        NOT = { has_global_flag = aiboost_initialized }
    }
    immediate = {
        set_global_flag = aiboost_initialized
        every_country = {
            limit = {
                is_ai = yes
                is_country_type = default
            }
            hidden_effect = {
                add_modifier = {
                    modifier = "aiboost_resource_modifier_00"
                    days = 3600
                }
            }
        }
        country_event = { id = aiboost.10 days = 3600 }
    }
}
